Crear tablas:
clubs
CREATE TABLE clubs (
    id SERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    city TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

players (incluye fantasmas)
CREATE TABLE players (
    id SERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    club_id INT REFERENCES clubs(id),
    is_registered BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT NOW()
);

club_events (🔥 clave)
CREATE TABLE club_events (
    id SERIAL PRIMARY KEY,
    club_id INT REFERENCES clubs(id),
    name TEXT NOT NULL,
    event_type TEXT, -- americano, cuadrangular, hexagonal, libre
    category TEXT,   -- D, C, B, 4ta
    gender TEXT,     -- femenino, masculino, mixto
    event_date DATE,
    status TEXT DEFAULT 'active',
    created_at TIMESTAMP DEFAULT NOW()
);


matches
CREATE TABLE matches (
    id SERIAL PRIMARY KEY,
    club_id INT REFERENCES clubs(id),
    event_id INT REFERENCES club_events(id), -- NULL = partido libre
    match_type TEXT DEFAULT 'standard', -- standard, americano_short, etc
    status TEXT DEFAULT 'pending', 
    created_by INT REFERENCES players(id),
    played_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW()
);

match_players
CREATE TABLE match_players (
    id SERIAL PRIMARY KEY,
    match_id INT REFERENCES matches(id),
    player_id INT REFERENCES players(id),
    team TEXT -- 'A' o 'B'
);


match_results
CREATE TABLE match_results (
    id SERIAL PRIMARY KEY,
    match_id INT REFERENCES matches(id),
    score TEXT -- ejemplo: "6-4 6-3"
);


match_confirmations
CREATE TABLE match_confirmations (
    id SERIAL PRIMARY KEY,
    match_id INT REFERENCES matches(id),
    player_id INT REFERENCES players(id),
    confirmed BOOLEAN,
    created_at TIMESTAMP DEFAULT NOW()
);


Probar que funciona
Inserta un club:
INSERT INTO clubs (name, city)
VALUES ('Espacio Active', 'Santiago');


Estados del partido
pending    → recién cargado
confirmed  → ambos lados validaron
disputed   → alguien rechazó
approved   → club validó
rejected   → club lo anuló


Peso de partidos (muy importante)
standard → peso 1.0
americano_short → peso 0.5
hexagonal → 0.75
cuadrangular → 1.0
